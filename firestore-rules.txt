  rules_version = '2';
  service cloud.firestore {
    match /databases/{database}/documents {
      // Users can read/write their own user document
      match /users/{userId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      
      // Candidates: public read (companies browse), owners write
      match /candidates/{document} {
        allow read: if true;
        allow create, update, delete: if request.auth != null && request.auth.uid == document;
      }
      
      // Subscriptions: owner read/write
      match /subscriptions/{userId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      
      // Purchases: companies read/write their purchases
      match /purchases/{purchaseId} {
        allow read, write: if request.auth != null;
        allow create: if request.auth != null && 
          request.resource.data.companyId == request.auth.uid;
      }
      
      // Admin dashboard access
      match /{document=**} {
        allow read, write: if request.auth != null && 
          exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
          get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin';
      }
    }
  }
