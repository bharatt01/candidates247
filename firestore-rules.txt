rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users can read/write their own profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Candidates: authenticated users can read all, only candidates can write their own
    match /candidates/{candidateId} {
      allow read: if true;  // Public read for browsing
      allow write: if request.auth != null && request.auth.uid == candidateId;
    }
    
    // Companies: authenticated users can read all, only companies can write their own
    match /companies/{companyId} {
      allow read: if true;  // Public read if needed
      allow write: if request.auth != null && request.auth.uid == companyId;
    }
    
    // Purchases: only the purchasing company can read/write their purchases
    match /purchases/{purchaseId} {
      allow read, write: if request.auth != null && 
        request.auth.uid == resource.data.companyUserId;
    }
  }
}
